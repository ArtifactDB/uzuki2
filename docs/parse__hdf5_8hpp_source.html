58template<class Function>
59void load_string_dataset(const H5::DataSet& handle, hsize_t full_length, Function fun) {
60 auto dtype = handle.getDataType();
61
62 // TODO: read this in chunks.
63 if (dtype.isVariableStr()) {
64 std::vector<char*> buffer(full_length);
65 handle.read(buffer.data(), dtype);
66
67 for (size_t i = 0; i < full_length; ++i) {
68 fun(i, std::string(buffer[i]));
69 }
70
71 auto dspace = handle.getSpace();
72 H5Dvlen_reclaim(dtype.getId(), dspace.getId(), H5P_DEFAULT, buffer.data());
73
74 } else {
75 size_t len = dtype.getSize();
76 std::vector<char> buffer(len * full_length);
77 handle.read(buffer.data(), dtype);
78
79 auto start = buffer.data();
80 for (size_t i = 0; i < full_length; ++i, start += len) {
81 size_t j = 0;
82 for (; j < len && start[j] != '\0'; ++j) {}
83 fun(i, std::string(start, start + j));
84 }
85 }

142template<class Host, class Function>
143void parse_integer_like(const H5::DataSet& handle, Host* ptr, const std::string& path, Function check, const Version& version) {
144 if (handle.getDataType().getClass() != H5T_INTEGER) {
145 throw std::runtime_error("expected an integer dataset at '" + path + "'");
146 }
147 forbid_large_integers(handle, path);
148
149 bool has_missing = false;
150 int32_t missing_value = -2147483648;
151 if (version.equals(1, 0)) {
152 has_missing = true;
153 } else {
154 has_missing = handle.attrExists("missing-value-placeholder");
155 if (has_missing) {
156 auto attr = handle.openAttribute("missing-value-placeholder");
157 if (attr.getTypeClass() != H5T_INTEGER || attr.getSpace().getSimpleExtentNdims() != 0) {
158 throw std::runtime_error("'missing-value-placeholder' attribute should be a scalar integer at '" + path + "'");
159 }
160 attr.read(H5::PredType::NATIVE_INT32, &missing_value);
161 }
162 }
163
164 // TODO: loop in chunks to reduce memory usage.
165 size_t len = ptr->size();
166 std::vector<int32_t> buffer(len);
167 handle.read(buffer.data(), H5::PredType::NATIVE_INT32);
168 for (hsize_t i = 0; i < len; ++i) {
169 auto current = buffer[i];
170 if (has_missing && current == missing_value) {
171 ptr->set_missing(i);
172 } else {
173 check(current);
174 ptr->set(i, current);
175 }
176 }

204inline double legacy_missing_double() {
205 uint32_t tmp_value = 1;
206 auto tmp_ptr = reinterpret_cast<unsigned char*>(&tmp_value);
207
208 // Mimic R's generation of these values, but we can't use type punning as
209 // this is not legal in C++, and we don't have bit_cast yet.
210 double missing_value = 0;
211 auto missing_ptr = reinterpret_cast<unsigned char*>(&missing_value);
212
213 int step = 1;
214 if (tmp_ptr[0] == 1) { // little-endian.
215 missing_ptr += sizeof(double) - 1;
216 step = -1;
217 }
218
219 *missing_ptr = 0x7f;
220 *(missing_ptr += step) = 0xf0;
221 *(missing_ptr += step) = 0x00;
222 *(missing_ptr += step) = 0x00;
223 *(missing_ptr += step) = 0x00;
224 *(missing_ptr += step) = 0x00;
225 *(missing_ptr += step) = 0x07;
226 *(missing_ptr += step) = 0xa2;
227
228 return missing_value;
229}

231template<class Host, class Function>
232void parse_numbers(const H5::DataSet& handle, Host* ptr, const std::string& path, Function check, const Version& version) {
233 if (handle.getDataType().getClass() != H5T_FLOAT) {
234 throw std::runtime_error("expected a float dataset at '" + path + "'");
235 }
236
237 H5::FloatType ftype(handle);
238 if (ftype.getPrecision() > 64) {
239 throw std::runtime_error("data type is potentially out of range for a double at '" + path + "'");
240 }
241
242 bool has_missing = false;
243 double missing_value = 0;
244
245 if (version.equals(1, 0)) {
246 has_missing = true;
247 missing_value = legacy_missing_double();
248 } else {
249 has_missing = handle.attrExists("missing-value-placeholder");
250 if (has_missing) {
251 auto attr = handle.openAttribute("missing-value-placeholder");
252 if (attr.getTypeClass() != H5T_FLOAT || attr.getSpace().getSimpleExtentNdims() != 0) {
253 throw std::runtime_error("'missing-value-placeholder' attribute should be a scalar float at '" + path + "'");
254 }
255 attr.read(H5::PredType::NATIVE_DOUBLE, &missing_value);
256 }
257 }
258
259 auto missing_ptr = reinterpret_cast<unsigned char*>(&missing_value);
260 auto is_missing = [&](double val) {
261 // Can't compare directly as missing_value or val might be NaN,
262 // so instead we need to do the comparison byte-by-byte.
263 auto candidate_ptr = reinterpret_cast<unsigned char*>(&val);
264 return std::memcmp(candidate_ptr, missing_ptr, sizeof(double)) == 0;
265 };
266
267 // TODO: loop in chunks to reduce memory usage.
268 size_t len = ptr->size();
269 std::vector<double> buffer(len);
270 handle.read(buffer.data(), H5::PredType::NATIVE_DOUBLE);
271 for (hsize_t i = 0; i < len; ++i) {
272 auto current = buffer[i];
273 if (has_missing && is_missing(current)) {
274 ptr->set_missing(i);
275 } else {
276 check(current);
277 ptr->set(i, current);
278 }
279 }

304template<class Provisioner, class Externals>
305std::shared_ptr<Base> parse_inner(const H5::Group& handle, Externals& ext, const std::string& path, const Version& version) {
306 // Deciding what type we're dealing with.
307 auto object_type = load_string_attribute(handle, "uzuki_object", path);
308 std::shared_ptr<Base> output;
309
310 if (object_type == "list") {
311 auto dpath = path + "/data";
312 if (!handle.exists("data") || handle.childObjType("data") != H5O_TYPE_GROUP) {
313 throw std::runtime_error("expected a group at '" + dpath + "'");
314 }
315 auto dhandle = handle.openGroup("data");
316 size_t len = dhandle.getNumObjs();
317
318 bool named = handle.exists("names");
319 auto lptr = Provisioner::new_List(len, named);
320 output.reset(lptr);
321
322 for (size_t i = 0; i < len; ++i) {
323 auto istr = std::to_string(i);
324 auto ipath = dpath + "/" + istr;
325 if (!dhandle.exists(istr) || dhandle.childObjType(istr) != H5O_TYPE_GROUP) {
326 throw std::runtime_error("expected a group at '" + ipath + "'");
327 }
328 auto lhandle = dhandle.openGroup(istr);
329 lptr->set(i, parse_inner<Provisioner>(lhandle, ext, ipath, version));
330 }
331
332 if (named) {
333 extract_names(handle, lptr, path, dpath);
334 }
335
336 } else if (object_type == "vector") {
337 auto vector_type = load_string_attribute(handle, "uzuki_type", path);
338
339 auto dpath = path + "/data";
340 if (!handle.exists("data") || handle.childObjType("data") != H5O_TYPE_DATASET) {
341 throw std::runtime_error("expected a dataset at '" + dpath + "'");
342 }
343 auto dhandle = handle.openDataSet("data");
344
345 auto len = check_1d_length(dhandle, dpath, true);
346 bool is_scalar = (len == 0);
347 if (is_scalar) {
348 len = 1;
349 }
350
351 bool named = handle.exists("names");
352
353 if (vector_type == "integer") {
354 auto iptr = Provisioner::new_Integer(len, named, is_scalar);
355 output.reset(iptr);
356 parse_integer_like(dhandle, iptr, dpath, [](int32_t) -> void {}, version);
357
358 } else if (vector_type == "boolean") {
359 auto bptr = Provisioner::new_Boolean(len, named, is_scalar);
360 output.reset(bptr);
361 parse_integer_like(dhandle, bptr, dpath, [&](int32_t x) -> void {
362 if (x != 0 && x != 1) {
363 throw std::runtime_error("boolean values should be 0 or 1 in '" + dpath + "'");
364 }
365 }, version);
366
367 } else if (vector_type == "factor" || (version.equals(1, 0) && vector_type == "ordered")) {
368 auto levpath = path + "/levels";
369 if (!handle.exists("levels") || handle.childObjType("levels") != H5O_TYPE_DATASET) {
370 throw std::runtime_error("expected a dataset at '" + levpath + "'");
371 }
372 auto levhandle = handle.openDataSet("levels");
373 auto levtype = levhandle.getDataType();
374 if (levtype.getClass() != H5T_STRING) {
375 throw std::runtime_error("expected a string dataset at '" + levpath + "'");
376 }
377 int32_t levlen = check_1d_length(levhandle, levpath, false); // use int-32 for comparison with the integer codes.
378
379 bool ordered = false;
380 if (vector_type == "ordered") {
381 ordered = true;
382 } else if (handle.exists("ordered")) {
383 auto ohandle = get_scalar_dataset(handle, "ordered", H5T_INTEGER, path);
384 int tmp_ordered = 0;
385 ohandle.read(&tmp_ordered, H5::PredType::NATIVE_INT);
386 ordered = tmp_ordered > 0;
387 }
388
389 auto fptr = Provisioner::new_Factor(len, named, is_scalar, levlen, ordered);
390 output.reset(fptr);
391 parse_integer_like(dhandle, fptr, dpath, [&](int32_t x) -> void {
392 if (x < 0 || x >= levlen) {
393 throw std::runtime_error("factor codes should be non-negative and less than the number of levels in '" + dpath + "'");
394 }
395 }, version);
396
397 std::unordered_set<std::string> present;
398 load_string_dataset(levhandle, levlen, [&](size_t i, std::string x) -> void {
399 if (present.find(x) != present.end()) {
400 throw std::runtime_error("levels should be unique at '" + levpath + "'");
401 }
402 fptr->set_level(i, x);
403 present.insert(x);
404 });
405
406 } else if (vector_type == "string" || (version.equals(1, 0) && (vector_type == "date" || vector_type == "date-time"))) {
407 StringVector::Format format = StringVector::NONE;
408 if (version.equals(1, 0)) {
409 if (vector_type == "date") {
410 format = StringVector::DATE;
411 } else if (vector_type == "date-time") {
412 format = StringVector::DATETIME;
413 }
414 } else if (handle.exists("format")) {
415 auto fhandle = get_scalar_dataset(handle, "format", H5T_STRING, path);
416 load_string_dataset(fhandle, 1, [&](size_t, std::string x) -> void {
417 if (x == "date") {
418 format = StringVector::DATE;
419 } else if (x == "date-time") {
420 format = StringVector::DATETIME;
421 } else {
422 throw std::runtime_error("unsupported format '" + x + "' at '" + path + "/format'");
423 }
424 });
425 }
426
427 auto sptr = Provisioner::new_String(len, named, is_scalar, format);
428 output.reset(sptr);
429 if (format == StringVector::NONE) {
430 parse_string_like(dhandle, sptr, dpath, [](const std::string&) -> void {});
431 } else if (format == StringVector::DATE) {
432 parse_string_like(dhandle, sptr, dpath, [&](const std::string& x) -> void {
433 if (!is_date(x)) {
434 throw std::runtime_error("dates should follow YYYY-MM-DD formatting in '" + path + ".values'");
435 }
436 });
437 } else if (format == StringVector::DATETIME) {
438 parse_string_like(dhandle, sptr, dpath, [&](const std::string& x) -> void {
439 if (!is_rfc3339(x)) {
440 throw std::runtime_error("date-times should follow the Internet Date/Time format in '" + path + ".values'");
441 }
442 });
443 }
444
445 } else if (vector_type == "number") {
446 auto dptr = Provisioner::new_Number(len, named, is_scalar);
447 output.reset(dptr);
448 parse_numbers(dhandle, dptr, dpath, [](double) -> void {}, version);
449
450 } else {
451 throw std::runtime_error("unknown vector type '" + vector_type + "' for '" + path + "'");
452 }
453
454 if (named) {
455 auto vptr = static_cast<Vector*>(output.get());
456 extract_names(handle, vptr, path, dpath);
457 }
458
459 } else if (object_type == "nothing") {
460 output.reset(Provisioner::new_Nothing());
461
462 } else if (object_type == "external") {
463 auto ipath = path + "/index";
464 if (!handle.exists("index") || handle.childObjType("index") != H5O_TYPE_DATASET) {
465 throw std::runtime_error("expected a dataset at '" + ipath + "'");
466 }
467
468 auto ihandle = handle.openDataSet("index");
469 if (ihandle.getDataType().getClass() != H5T_INTEGER) {
470 throw std::runtime_error("expected integer dataset at '" + ipath + "'");
471 }
472 forbid_large_integers(ihandle, ipath);
473
474 auto ispace = ihandle.getSpace();
475 int idims = ispace.getSimpleExtentNdims();
476 if (idims != 0) {
477 throw std::runtime_error("expected scalar dataset at '" + ipath + "'");
478 }
479
480 int idx;
481 ihandle.read(&idx, H5::PredType::NATIVE_INT);
482 if (idx < 0 || static_cast<size_t>(idx) >= ext.size()) {
483 throw std::runtime_error("external index out of range at '" + ipath + "'");
484 }
485
486 output.reset(Provisioner::new_External(ext.get(idx)));
487
488 } else {
489 throw std::runtime_error("unknown uzuki2 object type '" + object_type + "'");
490 }
491
492 return output;
493}
544template<class Provisioner, class Externals>
545std::shared_ptr<Base> parse(const H5::Group& handle, const std::string& name, Externals ext) {
546 Version version;
547 if (handle.attrExists("uzuki_version")) {
548 auto ver_str = load_string_attribute(handle.openAttribute("uzuki_version"), "uzuki_version", name);
549 version = parse_version_string(ver_str);
550 }
551
552 ExternalTracker etrack(std::move(ext));
553 auto ptr = parse_inner<Provisioner>(handle, etrack, name, version);
554 etrack.validate();
555 return ptr;
