55template<class Host, class Function>
56void parse_integer_like(const H5::DataSet& handle, Host* ptr, Function check, const Version& version, hsize_t buffer_size) try {
57 if (ritsuko::hdf5::exceeds_integer_limit(handle, 32, true)) {
58 throw std::runtime_error("dataset cannot be represented by 32-bit signed integers");
59 }
60
61 bool has_missing = false;
62 int32_t missing_value = -2147483648;
63 if (version.equals(1, 0)) {
64 has_missing = true;
65 } else {
66 const char* placeholder_name = "missing-value-placeholder";
67 has_missing = handle.attrExists(placeholder_name);
68 if (has_missing) {
69 auto attr = handle.openAttribute(placeholder_name);
70 ritsuko::hdf5::check_missing_placeholder_attribute(handle, attr, /* type_class_only = */ version.lt(1, 2));
71 attr.read(H5::PredType::NATIVE_INT32, &missing_value);
72 }
73 }
74
75 hsize_t full_length = ptr->size();
76 ritsuko::hdf5::Stream1dNumericDataset<int32_t> stream(&handle, full_length, buffer_size);
77 for (hsize_t i = 0; i < full_length; ++i, stream.next()) {
78 auto current = stream.get();
79 if (has_missing && current == missing_value) {
80 ptr->set_missing(i);
81 } else {
82 check(current);
83 ptr->set(i, current);
84 }
85 }
86
87} catch (std::exception& e) {
88 throw std::runtime_error("failed to load integer dataset at '" + ritsuko::hdf5::get_name(handle) + "'; " + std::string(e.what()));
89}

118template<class Host, class Function>
119void parse_numbers(const H5::DataSet& handle, Host* ptr, Function check, const Version& version, hsize_t buffer_size) try {
120 if (version.lt(1, 3)) {
121 if (handle.getTypeClass() != H5T_FLOAT) {
122 throw std::runtime_error("expected a floating-point dataset");
123 }
124 } else {
125 if (ritsuko::hdf5::exceeds_float_limit(handle, 64)) {
126 throw std::runtime_error("dataset cannot be represented by 64-bit floats");
127 }
128 }
129
130 bool has_missing = false;
131 double missing_value = 0;
132
133 if (version.equals(1, 0)) {
134 has_missing = true;
135 missing_value = ritsuko::r_missing_value();
136 } else {
137 const char* placeholder_name = "missing-value-placeholder";
138 has_missing = handle.attrExists(placeholder_name);
139 if (has_missing) {
140 auto attr = handle.openAttribute(placeholder_name);
141 ritsuko::hdf5::check_missing_placeholder_attribute(handle, attr, /* type_class_only = */ version.lt(1, 2));
142 attr.read(H5::PredType::NATIVE_DOUBLE, &missing_value);
143 }
144 }
145
146 bool should_compare_nan = version.lt(1, 3);
147 bool is_placeholder_nan = std::isnan(missing_value);
148 auto is_missing_value = [&](double val) -> bool {
149 if (should_compare_nan) {
150 return ritsuko::are_floats_identical(&val, &missing_value);
151 } else if (is_placeholder_nan) {
152 return std::isnan(val);
153 } else {
154 return val == missing_value;
155 }
156 };
157
158 hsize_t full_length = ptr->size();
159 ritsuko::hdf5::Stream1dNumericDataset<double> stream(&handle, full_length, buffer_size);
160 for (hsize_t i = 0; i < full_length; ++i, stream.next()) {
161 auto current = stream.get();
162 if (has_missing && is_missing_value(current)) {
163 ptr->set_missing(i);
164 } else {
165 check(current);
166 ptr->set(i, current);
167 }
168 }
169
170} catch (std::exception& e) {
171 throw std::runtime_error("failed to load floating-point dataset at '" + ritsuko::hdf5::get_name(handle) + "'; " + std::string(e.what()));
172}

204 std::shared_ptr<Base> output;
205
206 if (object_type == "list") {
207 auto dhandle = ritsuko::hdf5::open_group(handle, "data");
208 size_t len = dhandle.getNumObjs();
209
210 bool named = handle.exists("names");
211 auto lptr = Provisioner::new_List(len, named);
212 output.reset(lptr);
213
214 try {
215 for (size_t i = 0; i < len; ++i) {
216 auto istr = std::to_string(i);
217 auto lhandle = ritsuko::hdf5::open_group(dhandle, istr.c_str());
218 lptr->set(i, parse_inner<Provisioner>(lhandle, ext, version, buffer_size));
219 }
220 } catch (std::exception& e) {
221 throw std::runtime_error("failed to parse list contents in 'data'; " + std::string(e.what()));
222 }
223
224 if (named) {
225 extract_names(handle, lptr, buffer_size);
226 }
227
228 } else if (object_type == "vector") {
229 auto vector_type = ritsuko::hdf5::open_and_load_scalar_string_attribute(handle, "uzuki_type");
230
231 auto dhandle = ritsuko::hdf5::open_dataset(handle, "data");
232 size_t len = ritsuko::hdf5::get_1d_length(dhandle.getSpace(), true);
233 bool is_scalar = (len == 0);
234 if (is_scalar) {
235 len = 1;
236 }
237
238 bool named = handle.exists("names");
239
240 if (vector_type == "integer") {
241 auto iptr = Provisioner::new_Integer(len, named, is_scalar);
242 output.reset(iptr);
243 parse_integer_like(dhandle, iptr, [](int32_t) -> void {}, version, buffer_size);
244
245 } else if (vector_type == "boolean") {
246 auto bptr = Provisioner::new_Boolean(len, named, is_scalar);
247 output.reset(bptr);
248 parse_integer_like(dhandle, bptr, [&](int32_t x) -> void {
249 if (x != 0 && x != 1) {
250 throw std::runtime_error("boolean values should be 0 or 1");
251 }
252 }, version, buffer_size);
253
254 } else if (vector_type == "factor" || (version.equals(1, 0) && vector_type == "ordered")) {
255 auto levhandle = ritsuko::hdf5::open_dataset(handle, "levels");
256 auto levtype = levhandle.getDataType();
257 if (levtype.getClass() != H5T_STRING) {
258 throw std::runtime_error("expected a string dataset for the levels at 'levels'");
259 }
260
261 int32_t levlen = ritsuko::hdf5::get_1d_length(levhandle.getSpace(), false);
262 bool ordered = false;
263 if (vector_type == "ordered") {
264 ordered = true;
265 } else if (handle.exists("ordered")) {
266 auto ohandle = check_scalar_dataset(handle, "ordered");
267 if (ritsuko::hdf5::exceeds_integer_limit(ohandle, 32, true)) {
268 throw std::runtime_error("'ordered' value cannot be represented by a 32-bit integer");
269 }
270 int32_t tmp_ordered = 0;
271 ohandle.read(&tmp_ordered, H5::PredType::NATIVE_INT32);
272 ordered = tmp_ordered > 0;
273 }
274
275 auto fptr = Provisioner::new_Factor(len, named, is_scalar, levlen, ordered);
276 output.reset(fptr);
277 parse_integer_like(dhandle, fptr, [&](int32_t x) -> void {
278 if (x < 0 || x >= levlen) {
279 throw std::runtime_error("factor codes should be non-negative and less than the number of levels");
280 }
281 }, version, buffer_size);
282
283 std::unordered_set<std::string> present;
284 ritsuko::hdf5::Stream1dStringDataset stream(&levhandle, levlen, buffer_size);
285 for (int32_t i = 0; i < levlen; ++i, stream.next()) {
286 auto x = stream.steal();
287 if (present.find(x) != present.end()) {
288 throw std::runtime_error("levels should be unique");
289 }
290 fptr->set_level(i, x);
291 present.insert(std::move(x));
292 }
293
294 } else if (vector_type == "string" || (version.equals(1, 0) && (vector_type == "date" || vector_type == "date-time"))) {
295 StringVector::Format format = StringVector::NONE;
296 if (version.equals(1, 0)) {
297 if (vector_type == "date") {
298 format = StringVector::DATE;
299 } else if (vector_type == "date-time") {
300 format = StringVector::DATETIME;
301 }
302
303 } else if (handle.exists("format")) {
304 auto fhandle = check_scalar_dataset(handle, "format");
305 if (fhandle.getTypeClass() != H5T_STRING) {
306 throw std::runtime_error("'format' dataset should have a string datatype class");
307 }
308 auto x = ritsuko::hdf5::load_scalar_string_dataset(fhandle);
309 if (x == "date") {
310 format = StringVector::DATE;
311 } else if (x == "date-time") {
312 format = StringVector::DATETIME;
313 } else {
314 throw std::runtime_error("unsupported format '" + x + "'");
315 }
316 }
317

319 output.reset(sptr);
320 if (format == StringVector::NONE) {
321 parse_string_like(dhandle, sptr, [](const std::string&) -> void {}, buffer_size);
322
323 } else if (format == StringVector::DATE) {
324 parse_string_like(dhandle, sptr, [&](const std::string& x) -> void {
325 if (!ritsuko::is_date(x.c_str(), x.size())) {
326 throw std::runtime_error("dates should follow YYYY-MM-DD formatting");
327 }
328 }, buffer_size);
329
330 } else if (format == StringVector::DATETIME) {
331 parse_string_like(dhandle, sptr, [&](const std::string& x) -> void {
332 if (!ritsuko::is_rfc3339(x.c_str(), x.size())) {
333 throw std::runtime_error("date-times should follow the Internet Date/Time format");
334 }
335 }, buffer_size);
336 }
337
338 } else if (vector_type == "number") {
339 auto dptr = Provisioner::new_Number(len, named, is_scalar);
340 output.reset(dptr);
341 parse_numbers(dhandle, dptr, [](double) -> void {}, version, buffer_size);
342
343 } else {
344 throw std::runtime_error("unknown vector type '" + vector_type + "'");
345 }
346
347 if (named) {
348 auto vptr = static_cast<Vector*>(output.get());
349 extract_names(handle, vptr, buffer_size);
350 }
351
352 } else if (object_type == "nothing") {
353 output.reset(Provisioner::new_Nothing());
354
355 } else if (object_type == "external") {
356 auto ihandle = ritsuko::hdf5::open_dataset(handle, "index");
357 if (ritsuko::hdf5::exceeds_integer_limit(ihandle, 32, true)) {
358 throw std::runtime_error("external index at 'index' cannot be represented by a 32-bit signed integer");
359 }
360
361 auto ispace = ihandle.getSpace();
362 int idims = ispace.getSimpleExtentNdims();
363 if (idims != 0) {
364 throw std::runtime_error("expected scalar dataset at 'index'");
365 }
366
367 int32_t idx;
368 ihandle.read(&idx, H5::PredType::NATIVE_INT32);
369 if (idx < 0 || static_cast<size_t>(idx) >= ext.size()) {
370 throw std::runtime_error("external index out of range at 'index'");
371 }
372
373 output.reset(Provisioner::new_External(ext.get(idx)));
374
375 } else {
376 throw std::runtime_error("unknown uzuki2 object type '" + object_type + "'");
377 }
378
379 return output;
380} catch (std::exception& e) {
381 throw std::runtime_error("failed to load object at '" + ritsuko::hdf5::get_name(handle) + "'; " + std::string(e.what()));

447template<class Provisioner, class Externals>
448ParsedList parse(const H5::Group& handle, Externals ext, Options options = Options()) {
449 Version version;
450 if (handle.attrExists("uzuki_version")) {
451 auto ver_str = ritsuko::hdf5::open_and_load_scalar_string_attribute(handle, "uzuki_version");
452 auto vraw = ritsuko::parse_version_string(ver_str.c_str(), ver_str.size(), /* skip_patch = */ true);
453 version.major = vraw.major;
454 version.minor = vraw.minor;
455 }
456
457 ExternalTracker etrack(std::move(ext));
458 auto ptr = parse_inner<Provisioner>(handle, etrack, version, options.buffer_size);
459
460 if (options.strict_list && ptr->type() != LIST) {
461 throw std::runtime_error("top-level object should represent an R list");
462 }
463 etrack.validate();
464
465 return ParsedList(std::move(ptr), std::move(version));
466}
