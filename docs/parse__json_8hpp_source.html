34inline const std::vector<std::shared_ptr<millijson::Base> >& extract_array(
35 const std::unordered_map<std::string, std::shared_ptr<millijson::Base> >& properties,
36 const std::string& name,
37 const std::string& path)
38{
39 auto vIt = properties.find(name);
40 if (vIt == properties.end()) {
41 throw std::runtime_error("expected '" + name + "' property for object at '" + path + "'");

44 const auto& values_ptr = vIt->second;
45 if (values_ptr->type() != millijson::ARRAY) {

73template<class Destination, class Function>
74void extract_integers(const std::vector<std::shared_ptr<millijson::Base> >& values, Destination* dest, Function check, const std::string& path) {
75 for (size_t i = 0; i < values.size(); ++i) {
76 if (values[i]->type() == millijson::NOTHING) {
77 dest->set_missing(i);
78 continue;
79 }
80
81 if (values[i]->type() != millijson::NUMBER) {
82 throw std::runtime_error("expected a number at '" + path + ".values[" + std::to_string(i) + "]'");
83 }
84
85 auto val = static_cast<const millijson::Number*>(values[i].get())->value;
86 if (val != std::floor(val)) {
87 throw std::runtime_error("expected an integer at '" + path + ".values[" + std::to_string(i) + "]'");
88 }
89
90 constexpr double upper = std::numeric_limits<int32_t>::max();
91 constexpr double lower = std::numeric_limits<int32_t>::min();
92 if (val < lower || val > upper) {
93 throw std::runtime_error("value at '" + path + ".values[" + std::to_string(i) + "]' cannot be represented by a 32-bit signed integer");
94 }
95
96 int32_t ival = val;
97 if (val == -2147483648) {
98 dest->set_missing(i);
99 continue;
100 }
101
102 check(ival);
103 dest->set(i, ival);
104 }

125template<class Destination>
126void extract_names(const std::unordered_map<std::string, std::shared_ptr<millijson::Base> >& properties, Destination* dest, const std::string& path) {
127 auto nIt = properties.find("names");
128 if (nIt == properties.end()) {
129 return;
130 }
131
132 const auto name_ptr = nIt->second;
133 if (name_ptr->type() != millijson::ARRAY) {
134 throw std::runtime_error("expected an array in '" + path + ".names'");
135 }
136
137 const auto& names = static_cast<const millijson::Array*>(name_ptr.get())->values;
138 if (names.size() != dest->size()) {
139 throw std::runtime_error("length of 'names' and 'values' should be the same in '" + path + "'");
140 }
141 dest->use_names();
142
143 for (size_t i = 0; i < names.size(); ++i) {
144 if (names[i]->type() != millijson::STRING) {
145 throw std::runtime_error("expected a string at '" + path + ".names[" + std::to_string(i) + "]'");
146 }
147 dest->set_name(i, static_cast<const millijson::String*>(names[i].get())->value);
148 }

151template<class Provisioner, class Externals>
152std::shared_ptr<Base> parse_object(const millijson::Base* contents, Externals& ext, const std::string& path) {
153 if (contents->type() != millijson::OBJECT) {
154 throw std::runtime_error("each R object should be represented by a JSON object at '" + path + "'");
155 }
156
157 auto optr = static_cast<const millijson::Object*>(contents);
158 const auto& map = optr->values;
159
160 auto tIt = map.find("type");
161 if (tIt == map.end()) {
162 throw std::runtime_error("missing 'type' property for JSON object at '" + path + "'");
163 }
164 const auto& type_ptr = tIt->second;
165 if (type_ptr->type() != millijson::STRING) {
166 throw std::runtime_error("expected a string at '" + path + ".type'");
167 }
168 const auto& type = static_cast<const millijson::String*>(type_ptr.get())->value;
169
170 std::shared_ptr<Base> output;
171 if (type == "nothing") {
172 output.reset(Provisioner::new_Nothing());
173
174 } else if (type == "external") {
175 auto iIt = map.find("index");
176 if (iIt == map.end()) {
177 throw std::runtime_error("expected 'index' property for 'external' type at '" + path + "'");
178 }
179 const auto& index_ptr = iIt->second;
180 if (index_ptr->type() != millijson::NUMBER) {
181 throw std::runtime_error("expected a number at '" + path + ".index'");
182 }
183 auto index = static_cast<const millijson::Number*>(index_ptr.get())->value;
184
185 if (index != std::floor(index)) {
186 throw std::runtime_error("expected an integer at '" + path + ".index'");
187 } else if (index < 0 || index >= static_cast<double>(ext.size())) {
188 throw std::runtime_error("external index out of range at '" + path + ".index'");
189 }
190 output.reset(Provisioner::new_External(ext.get(index)));
191
192 } else if (type == "integer") {
193 process_array_or_scalar_values(map, path, [&](const auto& vals) -> auto {
194 auto ptr = Provisioner::new_Integer(vals.size());
195 output.reset(ptr);
196 extract_integers(vals, ptr, [](int32_t) -> void {}, path);
197 extract_names(map, ptr, path);
198 return ptr;
199 });
200
201 } else if (type == "factor" || type == "ordered") {
202 const auto& vals = extract_array(map, "values", path);
203 const auto& lvals = extract_array(map, "levels", path);
204
205 auto ptr = Provisioner::new_Factor(vals.size(), lvals.size());
206 output.reset(ptr);
207 if (type == "ordered") {
208 ptr->is_ordered();
209 }
210
211 int32_t nlevels = lvals.size();
212 extract_integers(vals, ptr, [&](int32_t x) -> void {
213 if (x < 0 || x >= nlevels) {
214 throw std::runtime_error("factor indices of out of range of levels in '" + path + "'");
215 }
216 }, path);
217
218 std::unordered_set<std::string> existing;
219 for (size_t l = 0; l < lvals.size(); ++l) {
220 if (lvals[l]->type() != millijson::STRING) {
221 throw std::runtime_error("expected strings at '" + path + ".levels[" + std::to_string(l) + "]'");
222 }
223
224 const auto& level = static_cast<const millijson::String*>(lvals[l].get())->value;
225 if (existing.find(level) != existing.end()) {
226 throw std::runtime_error("detected duplicate string at '" + path + ".levels[" + std::to_string(l) + "]'");
227 }
228 ptr->set_level(l, level);
229 existing.insert(level);
230 }
231
232 extract_names(map, ptr, path);
233
234 } else if (type == "boolean") {
235 process_array_or_scalar_values(map, path, [&](const auto& vals) -> auto {
236 auto ptr = Provisioner::new_Boolean(vals.size());
237 output.reset(ptr);
238
239 for (size_t i = 0; i < vals.size(); ++i) {
240 if (vals[i]->type() == millijson::NOTHING) {
241 ptr->set_missing(i);
242 continue;
243 }
244
245 if (vals[i]->type() != millijson::BOOLEAN) {
246 throw std::runtime_error("expected a boolean at '" + path + ".values[" + std::to_string(i) + "]'");
247 }
248 ptr->set(i, static_cast<const millijson::Boolean*>(vals[i].get())->value);
249 }
250
251 extract_names(map, ptr, path);
252 return ptr;
253 });
254
255 } else if (type == "number") {
256 process_array_or_scalar_values(map, path, [&](const auto& vals) -> auto {
257 auto ptr = Provisioner::new_Number(vals.size());
258 output.reset(ptr);
259
260 for (size_t i = 0; i < vals.size(); ++i) {
261 if (vals[i]->type() == millijson::NOTHING) {
262 ptr->set_missing(i);
263 continue;
264 }
265
266 if (vals[i]->type() == millijson::NUMBER) {
267 ptr->set(i, static_cast<const millijson::Number*>(vals[i].get())->value);
268 } else if (vals[i]->type() == millijson::STRING) {
269 auto str = static_cast<const millijson::String*>(vals[i].get())->value;
270 if (str == "NaN") {
271 ptr->set(i, std::numeric_limits<double>::quiet_NaN());
272 } else if (str == "Inf") {
273 ptr->set(i, std::numeric_limits<double>::infinity());
274 } else if (str == "-Inf") {
275 ptr->set(i, -std::numeric_limits<double>::infinity());
276 } else {
277 throw std::runtime_error("unsupported string '" + str + "' at '" + path + ".values[" + std::to_string(i) + "]'");
278 }
279 } else {
280 throw std::runtime_error("expected a number at '" + path + ".values[" + std::to_string(i) + "]'");
281 }
282 }
283
284 extract_names(map, ptr, path);
285 return ptr;
286 });
287
288 } else if (type == "string") {
289 process_array_or_scalar_values(map, path, [&](const auto& vals) -> auto {
290 auto ptr = Provisioner::new_String(vals.size());
291 output.reset(ptr);
292 extract_strings(vals, ptr, [](const std::string&) -> void {}, path);
293 extract_names(map, ptr, path);
294 return ptr;
295 });
296
297 } else if (type == "date") {
298 process_array_or_scalar_values(map, path, [&](const auto& vals) -> auto {
299 auto ptr = Provisioner::new_Date(vals.size());
300 output.reset(ptr);
301 extract_strings(vals, ptr, [&](const std::string& x) -> void {
302 if (!is_date(x)) {
303 throw std::runtime_error("dates should follow YYYY-MM-DD formatting in '" + path + ".values'");
304 }
305 }, path);
306 extract_names(map, ptr, path);
307 return ptr;
308 });
309
310 } else if (type == "date-time") {
311 const auto& vals = extract_array(map, "values", path);
312 auto ptr = Provisioner::new_DateTime(vals.size());
313 output.reset(ptr);
314 extract_strings(vals, ptr, [&](const std::string& x) -> void {
315 if (!is_rfc3339(x)) {
316 throw std::runtime_error("date-times should follow the Internet Date/Time format in '" + path + ".values'");
317 }
318 }, path);
319 extract_names(map, ptr, path);
320
321 } else if (type == "list") {
322 const auto& vals = extract_array(map, "values", path);
323 auto ptr = Provisioner::new_List(vals.size());
324 output.reset(ptr);
325 for (size_t i = 0; i < vals.size(); ++i) {
326 ptr->set(i, parse_object<Provisioner>(vals[i].get(), ext, path + ".values[" + std::to_string(i) + "]"));
327 }
328 extract_names(map, ptr, path);
329
330 } else {
331 throw std::runtime_error("unknown object type '" + type + "' at '" + path + ".type'");
332 }
333
334 return output;
335}
