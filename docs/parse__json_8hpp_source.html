52template<class Destination, class Function>
53void extract_integers(const std::vector<std::shared_ptr<millijson::Base> >& values, Destination* dest, Function check, const std::string& path) {
54 for (size_t i = 0; i < values.size(); ++i) {
55 if (values[i]->type() == millijson::NOTHING) {
56 dest->set_missing(i);
57 continue;
58 }
59
60 if (values[i]->type() != millijson::NUMBER) {
61 throw std::runtime_error("expected a number at '" + path + ".values[" + std::to_string(i) + "]'");
62 }
63
64 auto val = static_cast<const millijson::Number*>(values[i].get())->value;
65 if (val != std::floor(val)) {
66 throw std::runtime_error("expected an integer at '" + path + ".values[" + std::to_string(i) + "]'");
67 }
68
69 constexpr double upper = std::numeric_limits<int32_t>::max();
70 constexpr double lower = std::numeric_limits<int32_t>::min();
71 if (val < lower || val > upper) {
72 throw std::runtime_error("value at '" + path + ".values[" + std::to_string(i) + "]' cannot be represented by a 32-bit signed integer");
73 }
74
75 int32_t ival = val;
76 check(ival);
77 dest->set(i, ival);
78 }

99template<class Destination>
100void extract_names(const std::unordered_map<std::string, std::shared_ptr<millijson::Base> >& properties, Destination* dest, const std::string& path) {
101 auto nIt = properties.find("names");
102 if (nIt == properties.end()) {
103 return;
104 }
105
106 const auto name_ptr = nIt->second;
107 if (name_ptr->type() != millijson::ARRAY) {
108 throw std::runtime_error("expected an array in '" + path + ".names'");
109 }
110
111 const auto& names = static_cast<const millijson::Array*>(name_ptr.get())->values;
112 if (names.size() != dest->size()) {
113 throw std::runtime_error("length of 'names' and 'values' should be the same in '" + path + "'");
114 }
115 dest->use_names();
116
117 for (size_t i = 0; i < names.size(); ++i) {
118 if (names[i]->type() != millijson::STRING) {
119 throw std::runtime_error("expected a string at '" + path + ".names[" + std::to_string(i) + "]'");
120 }
121 dest->set_name(i, static_cast<const millijson::String*>(names[i].get())->value);
122 }

125template<class Provisioner, class Externals>
126std::shared_ptr<Base> parse_object(const millijson::Base* contents, Externals& ext, const std::string& path) {
127 if (contents->type() != millijson::OBJECT) {
128 throw std::runtime_error("each R object should be represented by a JSON object at '" + path + "'");
129 }
130
131 auto optr = static_cast<const millijson::Object*>(contents);
132 const auto& map = optr->values;
133
134 auto tIt = map.find("type");
135 if (tIt == map.end()) {
136 throw std::runtime_error("missing 'type' property for JSON object at '" + path + "'");
137 }
138 const auto& type_ptr = tIt->second;
139 if (type_ptr->type() != millijson::STRING) {
140 throw std::runtime_error("expected a string at '" + path + ".type'");
141 }
142 const auto& type = static_cast<const millijson::String*>(type_ptr.get())->value;
143
144 std::shared_ptr<Base> output;
145 if (type == "nothing") {
146 output.reset(Provisioner::new_Nothing());
147
148 } else if (type == "external") {
149 auto iIt = map.find("index");
150 if (iIt == map.end()) {
151 throw std::runtime_error("expected 'index' property for 'external' type at '" + path + "'");
152 }
153 const auto& index_ptr = iIt->second;
154 if (index_ptr->type() != millijson::NUMBER) {
155 throw std::runtime_error("expected a number at '" + path + ".index'");
156 }
157 auto index = static_cast<const millijson::Number*>(index_ptr.get())->value;
158
159 if (index != std::floor(index)) {
160 throw std::runtime_error("expected an integer at '" + path + ".index'");
161 } else if (index < 0 || index >= static_cast<double>(ext.size())) {
162 throw std::runtime_error("external index out of range at '" + path + ".index'");
163 }
164 output.reset(Provisioner::new_External(ext.get(index)));
165
166 } else if (type == "integer") {
167 const auto& vals = extract_array(map, "values", path);
168 auto ptr = Provisioner::new_Integer(vals.size());
169 output.reset(ptr);
170 extract_integers(vals, ptr, [](int32_t) -> void {}, path);
171 extract_names(map, ptr, path);
172
173 } else if (type == "factor" || type == "ordered") {
174 const auto& vals = extract_array(map, "values", path);
175 const auto& lvals = extract_array(map, "levels", path);
176
177 auto ptr = Provisioner::new_Factor(vals.size(), lvals.size());
178 output.reset(ptr);
179 if (type == "ordered") {
180 ptr->is_ordered();
181 }
182
183 int32_t nlevels = lvals.size();
184 extract_integers(vals, ptr, [&](int32_t x) -> void {
185 if (x < 0 || x >= nlevels) {
186 throw std::runtime_error("factor indices of out of range of levels in '" + path + "'");
187 }
188 }, path);
189
190 std::unordered_set<std::string> existing;
191 for (size_t l = 0; l < lvals.size(); ++l) {
192 if (lvals[l]->type() != millijson::STRING) {
193 throw std::runtime_error("expected strings at '" + path + ".levels[" + std::to_string(l) + "]'");
194 }
195
196 const auto& level = static_cast<const millijson::String*>(lvals[l].get())->value;
197 if (existing.find(level) != existing.end()) {
198 throw std::runtime_error("detected duplicate string at '" + path + ".levels[" + std::to_string(l) + "]'");
199 }
200 ptr->set_level(l, level);
201 existing.insert(level);
202 }
203
204 extract_names(map, ptr, path);
205
206 } else if (type == "boolean") {
207 const auto& vals = extract_array(map, "values", path);
208 auto ptr = Provisioner::new_Boolean(vals.size());
209 output.reset(ptr);
210
211 for (size_t i = 0; i < vals.size(); ++i) {
212 if (vals[i]->type() == millijson::NOTHING) {
213 ptr->set_missing(i);
214 continue;
215 }
216
217 if (vals[i]->type() != millijson::BOOLEAN) {
218 throw std::runtime_error("expected a boolean at '" + path + ".values[" + std::to_string(i) + "]'");
219 }
220 ptr->set(i, static_cast<const millijson::Boolean*>(vals[i].get())->value);
221 }
222
223 extract_names(map, ptr, path);
224
225 } else if (type == "number") {
226 const auto& vals = extract_array(map, "values", path);
227 auto ptr = Provisioner::new_Number(vals.size());
228 output.reset(ptr);
229
230 for (size_t i = 0; i < vals.size(); ++i) {
231 if (vals[i]->type() == millijson::NOTHING) {
232 ptr->set_missing(i);
233 continue;
234 }
235
236 if (vals[i]->type() == millijson::NUMBER) {
237 ptr->set(i, static_cast<const millijson::Number*>(vals[i].get())->value);
238 } else if (vals[i]->type() == millijson::STRING) {
239 auto str = static_cast<const millijson::String*>(vals[i].get())->value;
240 if (str == "NaN") {
241 ptr->set(i, std::numeric_limits<double>::quiet_NaN());
242 } else if (str == "Inf") {
243 ptr->set(i, std::numeric_limits<double>::infinity());
244 } else if (str == "-Inf") {
245 ptr->set(i, -std::numeric_limits<double>::infinity());
246 } else {
247 throw std::runtime_error("unsupported string '" + str + "' at '" + path + ".values[" + std::to_string(i) + "]'");
248 }
249 } else {
250 throw std::runtime_error("expected a number at '" + path + ".values[" + std::to_string(i) + "]'");
251 }
252 }
253
254 extract_names(map, ptr, path);
255
256 } else if (type == "string") {
257 const auto& vals = extract_array(map, "values", path);
258 auto ptr = Provisioner::new_String(vals.size());
259 output.reset(ptr);
260 extract_strings(vals, ptr, [](const std::string&) -> void {}, path);
261 extract_names(map, ptr, path);
262
263 } else if (type == "date") {
264 const auto& vals = extract_array(map, "values", path);
265 auto ptr = Provisioner::new_Date(vals.size());
266 output.reset(ptr);
267 extract_strings(vals, ptr, [&](const std::string& x) -> void {
268 if (!is_date(x)) {
269 throw std::runtime_error("dates should follow YYYY-MM-DD formatting in '" + path + ".values'");
270 }
271 }, path);
272 extract_names(map, ptr, path);
273
274 } else if (type == "list") {
275 const auto& vals = extract_array(map, "values", path);
276 auto ptr = Provisioner::new_List(vals.size());
277 output.reset(ptr);
278 for (size_t i = 0; i < vals.size(); ++i) {
279 ptr->set(i, parse_object<Provisioner>(vals[i].get(), ext, path + ".values[" + std::to_string(i) + "]"));
280 }
281 extract_names(map, ptr, path);
282
283 } else {
284 throw std::runtime_error("unknown object type '" + type + "' at '" + path + ".type'");
285 }
286
287 return output;
288}
