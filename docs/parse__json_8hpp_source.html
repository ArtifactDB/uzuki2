89template<class Function_>
90auto process_array_or_scalar_values(
91 const std::unordered_map<std::string, std::shared_ptr<millijson::Base> >& properties,
92 const std::string& path,
93 Function_ fun)
94{
95 auto vIt = properties.find("values");
96 if (vIt == properties.end()) {
97 throw std::runtime_error("expected 'values' property for object at '" + path + "'");
98 }
99
100 auto names_ptr = has_names(properties, path);
101 bool has_names = names_ptr != NULL;
102
103 typename std::invoke_result<Function_,std::vector<std::shared_ptr<millijson::Base> >,bool,bool>::type out_ptr;
104
105 const auto& values_ptr = vIt->second;
106 if (values_ptr->type() == millijson::ARRAY) {
107 out_ptr = fun(static_cast<const millijson::Array*>(values_ptr.get())->values, has_names, false);
108 } else {
109 std::vector<std::shared_ptr<millijson::Base> > temp { values_ptr };
110 out_ptr = fun(temp, has_names, true);
111 }
112
113 if (has_names) {
114 fill_names(names_ptr, out_ptr, path);
115 }
116 return out_ptr;
117}

119template<class Destination_, class Function_>
120void extract_integers(const std::vector<std::shared_ptr<millijson::Base> >& values, Destination_* dest, Function_ check, const std::string& path, const Version& version) {
121 for (size_t i = 0; i < values.size(); ++i) {
122 if (values[i]->type() == millijson::NOTHING) {
123 dest->set_missing(i);
124 continue;
125 }
126
127 if (values[i]->type() != millijson::NUMBER) {
128 throw std::runtime_error("expected a number at '" + path + ".values[" + std::to_string(i) + "]'");
129 }
130
131 auto val = static_cast<const millijson::Number*>(values[i].get())->value;
132 if (val != std::floor(val)) {
133 throw std::runtime_error("expected an integer at '" + path + ".values[" + std::to_string(i) + "]'");
134 }
135
136 constexpr double upper = std::numeric_limits<int32_t>::max();
137 constexpr double lower = std::numeric_limits<int32_t>::min();
138 if (val < lower || val > upper) {
139 throw std::runtime_error("value at '" + path + ".values[" + std::to_string(i) + "]' cannot be represented by a 32-bit signed integer");
140 }
141
142 int32_t ival = val;
143 if (version.equals(1, 0) && val == -2147483648) {
144 dest->set_missing(i);
145 continue;
146 }
147
148 check(ival);
149 dest->set(i, ival);
150 }

172std::shared_ptr<Base> parse_object(const millijson::Base* contents, Externals_& ext, const std::string& path, const Version& version) {
173 if (contents->type() != millijson::OBJECT) {
174 throw std::runtime_error("each R object should be represented by a JSON object at '" + path + "'");
175 }
176
177 auto optr = static_cast<const millijson::Object*>(contents);
178 const auto& map = optr->values;
179
180 auto tIt = map.find("type");
181 if (tIt == map.end()) {
182 throw std::runtime_error("missing 'type' property for JSON object at '" + path + "'");
183 }
184 const auto& type_ptr = tIt->second;
185 if (type_ptr->type() != millijson::STRING) {
186 throw std::runtime_error("expected a string at '" + path + ".type'");
187 }
188 const auto& type = static_cast<const millijson::String*>(type_ptr.get())->value;
189
190 std::shared_ptr<Base> output;
191 if (type == "nothing") {
192 output.reset(Provisioner_::new_Nothing());
193
194 } else if (type == "external") {
195 auto iIt = map.find("index");
196 if (iIt == map.end()) {
197 throw std::runtime_error("expected 'index' property for 'external' type at '" + path + "'");
198 }
199 const auto& index_ptr = iIt->second;
200 if (index_ptr->type() != millijson::NUMBER) {
201 throw std::runtime_error("expected a number at '" + path + ".index'");
202 }
203 auto index = static_cast<const millijson::Number*>(index_ptr.get())->value;
204
205 if (index != std::floor(index)) {
206 throw std::runtime_error("expected an integer at '" + path + ".index'");
207 } else if (index < 0 || index >= static_cast<double>(ext.size())) {
208 throw std::runtime_error("external index out of range at '" + path + ".index'");
209 }
210 output.reset(Provisioner_::new_External(ext.get(index)));
211
212 } else if (type == "integer") {
213 process_array_or_scalar_values(map, path, [&](const auto& vals, bool named, bool scalar) -> auto {
214 auto ptr = Provisioner_::new_Integer(vals.size(), named, scalar);
215 output.reset(ptr);
216 extract_integers(vals, ptr, [](int32_t) -> void {}, path, version);
217 return ptr;
218 });
219
220 } else if (type == "factor" || (version.equals(1, 0) && type == "ordered")) {
221 bool ordered = false;
222 if (type == "ordered") {
223 ordered = true;
224 } else {
225 auto oIt = map.find("ordered");
226 if (oIt != map.end()) {
227 if (oIt->second->type() != millijson::BOOLEAN) {
228 throw std::runtime_error("expected a boolean at '" + path + ".ordered'");
229 }
230 auto optr = static_cast<const millijson::Boolean*>((oIt->second).get());
231 ordered = optr->value;
232 }
233 }

236 const auto& lvals = extract_array(map, levels_name, path);
237 int32_t nlevels = lvals.size();
238 auto fptr = process_array_or_scalar_values(map, path, [&](const auto& vals, bool named, bool scalar) -> auto {
239 auto ptr = Provisioner_::new_Factor(vals.size(), named, scalar, nlevels, ordered);
240 output.reset(ptr);
241 extract_integers(vals, ptr, [&](int32_t x) -> void {
242 if (x < 0 || x >= nlevels) {
243 throw std::runtime_error("factor indices of out of range of levels in '" + path + "'");
244 }
245 }, path, version);
246 return ptr;
247 });

253 }

265 auto ptr = Provisioner_::new_Boolean(vals.size(), named, scalar);
266 output.reset(ptr);

285 auto ptr = Provisioner_::new_Number(vals.size(), named, scalar);
286 output.reset(ptr);
287
288 for (size_t i = 0; i < vals.size(); ++i) {
289 if (vals[i]->type() == millijson::NOTHING) {
290 ptr->set_missing(i);
291 continue;

293

309 }
310 }

313 });

317 if (version.equals(1, 0)) {
318 if (type == "date") {
319 format = StringVector::DATE;
320 } else if (type == "date-time") {
321 format = StringVector::DATETIME;
322 }
323 } else {

325 if (fIt != map.end()) {
326 if (fIt->second->type() != millijson::STRING) {
327 throw std::runtime_error("expected a string at '" + path + ".format'");
328 }
329 auto fptr = static_cast<const millijson::String*>(fIt->second.get());
330 if (fptr->value == "date") {
331 format = StringVector::DATE;
332 } else if (fptr->value == "date-time") {
333 format = StringVector::DATETIME;
334 } else {
335 throw std::runtime_error("unsupported format '" + fptr->value + "' at '" + path + ".format'");
336 }
337 }
338 }

340 process_array_or_scalar_values(map, path, [&](const auto& vals, bool named, bool scalar) -> auto {
341 auto ptr = Provisioner_::new_String(vals.size(), named, scalar, format);
342 output.reset(ptr);
343
344 if (format == StringVector::NONE) {
345 extract_strings(vals, ptr, [](const std::string&) -> void {}, path);
346 } else if (format == StringVector::DATE) {
347 extract_strings(vals, ptr, [&](const std::string& x) -> void {
348 if (!ritsuko::is_date(x.c_str(), x.size())) {
349 throw std::runtime_error("dates should follow YYYY-MM-DD formatting in '" + path + ".values'");
350 }
351 }, path);
352 } else if (format == StringVector::DATETIME) {
353 extract_strings(vals, ptr, [&](const std::string& x) -> void {
354 if (!ritsuko::is_rfc3339(x.c_str(), x.size())) {
355 throw std::runtime_error("date-times should follow the Internet Date/Time format in '" + path + ".values'");
356 }
357 }, path);
358 }
359
360 return ptr;
361 });
362
363 } else if (type == "list") {
364 auto names_ptr = has_names(map, path);
365 bool has_names = names_ptr != NULL;
366
367 const std::string values_name = "values"; // avoid dangling reference from casting of string literal.
368 const auto& vals = extract_array(map, values_name, path);
369 auto ptr = Provisioner_::new_List(vals.size(), has_names);
370 output.reset(ptr);
371
372 for (size_t i = 0; i < vals.size(); ++i) {
373 ptr->set(i, parse_object<Provisioner_>(vals[i].get(), ext, path + ".values[" + std::to_string(i) + "]", version));
374 }
375
376 if (has_names) {
377 fill_names(names_ptr, ptr, path);
378 }
379
380 } else {
381 throw std::runtime_error("unknown object type '" + type + "' at '" + path + ".type'");
382 }
383
384 return output;

429template<class Provisioner_, class Externals_>
430ParsedList parse(byteme::Reader& reader, Externals_ ext, const Options& options) {
431 std::unique_ptr<byteme::PerByteInterface<char> > pb;
432 if (options.parallel) {
433 pb.reset(new byteme::PerByteSerial<char, byteme::Reader*>(&reader));
434 } else {
435 pb.reset(new byteme::PerByteParallel<char, byteme::Reader*>(&reader));
436 }
437 auto contents = millijson::parse(*pb);
438
439 Version version;
440 if (contents->type() == millijson::OBJECT) {
441 auto optr = static_cast<const millijson::Object*>(contents.get());
442 const auto& map = optr->values;
443 auto vIt = map.find("version");
444 if (vIt != map.end()) {
445 if (vIt->second->type() != millijson::STRING) {
446 throw std::runtime_error("expected a string in 'version'");
447 }
448 const auto& vstr = static_cast<const millijson::String*>(vIt->second.get())->value;
449 auto vraw = ritsuko::parse_version_string(vstr.c_str(), vstr.size(), /* skip_patch = */ true);
450 version.major = vraw.major;
451 version.minor = vraw.minor;
452 }
453 }
454
455 ExternalTracker etrack(std::move(ext));
456 auto output = parse_object<Provisioner_>(contents.get(), etrack, "", version);
457
458 if (options.strict_list && output->type() != LIST) {
459 throw std::runtime_error("top-level object should represent an R list");
460 }
461 etrack.validate();
462
463 return ParsedList(std::move(output), std::move(version));
464}
