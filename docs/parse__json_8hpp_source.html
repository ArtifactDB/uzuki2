74template<class Destination, class Function>
75void extract_integers(const std::vector<std::shared_ptr<millijson::Base> >& values, Destination* dest, Function check, const std::string& path, const Version& version) {
76 for (size_t i = 0; i < values.size(); ++i) {
77 if (values[i]->type() == millijson::NOTHING) {
78 dest->set_missing(i);
79 continue;
80 }
81
82 if (values[i]->type() != millijson::NUMBER) {
83 throw std::runtime_error("expected a number at '" + path + ".values[" + std::to_string(i) + "]'");
84 }
85
86 auto val = static_cast<const millijson::Number*>(values[i].get())->value;
87 if (val != std::floor(val)) {
88 throw std::runtime_error("expected an integer at '" + path + ".values[" + std::to_string(i) + "]'");
89 }
90
91 constexpr double upper = std::numeric_limits<int32_t>::max();
92 constexpr double lower = std::numeric_limits<int32_t>::min();
93 if (val < lower || val > upper) {
94 throw std::runtime_error("value at '" + path + ".values[" + std::to_string(i) + "]' cannot be represented by a 32-bit signed integer");
95 }
96
97 int32_t ival = val;
98 if (version.equals(1, 0) && val == -2147483648) {
99 dest->set_missing(i);
100 continue;
101 }
102
103 check(ival);
104 dest->set(i, ival);
105 }

126template<class Destination>
127void extract_names(const std::unordered_map<std::string, std::shared_ptr<millijson::Base> >& properties, Destination* dest, const std::string& path) {
128 auto nIt = properties.find("names");
129 if (nIt == properties.end()) {
130 return;
131 }
132
133 const auto name_ptr = nIt->second;
134 if (name_ptr->type() != millijson::ARRAY) {
135 throw std::runtime_error("expected an array in '" + path + ".names'");
136 }
137
138 const auto& names = static_cast<const millijson::Array*>(name_ptr.get())->values;
139 if (names.size() != dest->size()) {
140 throw std::runtime_error("length of 'names' and 'values' should be the same in '" + path + "'");
141 }
142 dest->use_names();
143
144 for (size_t i = 0; i < names.size(); ++i) {
145 if (names[i]->type() != millijson::STRING) {
146 throw std::runtime_error("expected a string at '" + path + ".names[" + std::to_string(i) + "]'");
147 }
148 dest->set_name(i, static_cast<const millijson::String*>(names[i].get())->value);
149 }

153std::shared_ptr<Base> parse_object(const millijson::Base* contents, Externals& ext, const std::string& path, const Version& version) {
154 if (contents->type() != millijson::OBJECT) {
155 throw std::runtime_error("each R object should be represented by a JSON object at '" + path + "'");
156 }
157
158 auto optr = static_cast<const millijson::Object*>(contents);
159 const auto& map = optr->values;
160
161 auto tIt = map.find("type");
162 if (tIt == map.end()) {
163 throw std::runtime_error("missing 'type' property for JSON object at '" + path + "'");
164 }
165 const auto& type_ptr = tIt->second;
166 if (type_ptr->type() != millijson::STRING) {
167 throw std::runtime_error("expected a string at '" + path + ".type'");
168 }
169 const auto& type = static_cast<const millijson::String*>(type_ptr.get())->value;
170
171 std::shared_ptr<Base> output;
172 if (type == "nothing") {
173 output.reset(Provisioner::new_Nothing());
174
175 } else if (type == "external") {
176 auto iIt = map.find("index");
177 if (iIt == map.end()) {
178 throw std::runtime_error("expected 'index' property for 'external' type at '" + path + "'");
179 }
180 const auto& index_ptr = iIt->second;
181 if (index_ptr->type() != millijson::NUMBER) {
182 throw std::runtime_error("expected a number at '" + path + ".index'");
183 }
184 auto index = static_cast<const millijson::Number*>(index_ptr.get())->value;
185
186 if (index != std::floor(index)) {
187 throw std::runtime_error("expected an integer at '" + path + ".index'");
188 } else if (index < 0 || index >= static_cast<double>(ext.size())) {
189 throw std::runtime_error("external index out of range at '" + path + ".index'");
190 }
191 output.reset(Provisioner::new_External(ext.get(index)));
192
193 } else if (type == "integer") {
194 process_array_or_scalar_values(map, path, [&](const auto& vals) -> auto {
195 auto ptr = Provisioner::new_Integer(vals.size());
196 output.reset(ptr);
197 extract_integers(vals, ptr, [](int32_t) -> void {}, path, version);
198 extract_names(map, ptr, path);
199 return ptr;
200 });
201
202 } else if (type == "factor" || (version.equals(1, 0) && type == "ordered")) {
203 const auto& vals = extract_array(map, "values", path);
204 const auto& lvals = extract_array(map, "levels", path);
205
206 auto ptr = Provisioner::new_Factor(vals.size(), lvals.size());
207 output.reset(ptr);
208 if (type == "ordered") {
209 ptr->is_ordered();
210 } else {
211 auto oIt = map.find("ordered");
212 if (oIt != map.end()) {
213 if (oIt->second->type() != millijson::BOOLEAN) {
214 throw std::runtime_error("expected a boolean at '" + path + ".ordered'");
215 }
216 auto optr = static_cast<const millijson::Boolean*>((oIt->second).get());
217 if (optr->value) {
218 ptr->is_ordered();
219 }
220 }
221 }
222
223 int32_t nlevels = lvals.size();
224 extract_integers(vals, ptr, [&](int32_t x) -> void {
225 if (x < 0 || x >= nlevels) {
226 throw std::runtime_error("factor indices of out of range of levels in '" + path + "'");
227 }
228 }, path, version);
229
230 std::unordered_set<std::string> existing;
231 for (size_t l = 0; l < lvals.size(); ++l) {
232 if (lvals[l]->type() != millijson::STRING) {
233 throw std::runtime_error("expected strings at '" + path + ".levels[" + std::to_string(l) + "]'");
234 }
235
236 const auto& level = static_cast<const millijson::String*>(lvals[l].get())->value;
237 if (existing.find(level) != existing.end()) {
238 throw std::runtime_error("detected duplicate string at '" + path + ".levels[" + std::to_string(l) + "]'");
239 }
240 ptr->set_level(l, level);
241 existing.insert(level);
242 }
243
244 extract_names(map, ptr, path);
245
246 } else if (type == "boolean") {
247 process_array_or_scalar_values(map, path, [&](const auto& vals) -> auto {
248 auto ptr = Provisioner::new_Boolean(vals.size());
249 output.reset(ptr);
250
251 for (size_t i = 0; i < vals.size(); ++i) {
252 if (vals[i]->type() == millijson::NOTHING) {
253 ptr->set_missing(i);
254 continue;
255 }
256
257 if (vals[i]->type() != millijson::BOOLEAN) {
258 throw std::runtime_error("expected a boolean at '" + path + ".values[" + std::to_string(i) + "]'");
259 }
260 ptr->set(i, static_cast<const millijson::Boolean*>(vals[i].get())->value);
261 }
262
263 extract_names(map, ptr, path);
264 return ptr;
265 });
266
267 } else if (type == "number") {
268 process_array_or_scalar_values(map, path, [&](const auto& vals) -> auto {
269 auto ptr = Provisioner::new_Number(vals.size());
270 output.reset(ptr);
271
272 for (size_t i = 0; i < vals.size(); ++i) {
273 if (vals[i]->type() == millijson::NOTHING) {
274 ptr->set_missing(i);
275 continue;
276 }
277
278 if (vals[i]->type() == millijson::NUMBER) {
279 ptr->set(i, static_cast<const millijson::Number*>(vals[i].get())->value);
280 } else if (vals[i]->type() == millijson::STRING) {
281 auto str = static_cast<const millijson::String*>(vals[i].get())->value;
282 if (str == "NaN") {

284 } else if (str == "Inf") {
285 ptr->set(i, std::numeric_limits<double>::infinity());
286 } else if (str == "-Inf") {
287 ptr->set(i, -std::numeric_limits<double>::infinity());
288 } else {

290 }
291 } else {
292 throw std::runtime_error("expected a number at '" + path + ".values[" + std::to_string(i) + "]'");
293 }
294 }

296 extract_names(map, ptr, path);
297 return ptr;
298 });
299
300 } else if (type == "string" || (version.equals(1, 0) && (type == "date" || type == "date-time"))) {
301 StringVector::Format format = StringVector::NONE;
302 if (version.equals(1, 0)) {
303 if (type == "date") {
304 format = StringVector::DATE;
305 } else if (type == "date-time") {
306 format = StringVector::DATETIME;
307 }
308 } else {
309 auto fIt = map.find("format");
310 if (fIt != map.end()) {
311 if (fIt->second->type() != millijson::STRING) {
312 throw std::runtime_error("expected a string at '" + path + ".format'");
313 }
314 auto fptr = static_cast<const millijson::String*>(fIt->second.get());
315 if (fptr->value == "date") {
316 format = StringVector::DATE;
317 } else if (fptr->value == "date-time") {
318 format = StringVector::DATETIME;
319 } else {
320 throw std::runtime_error("unsupported format '" + fptr->value + "' at '" + path + ".format'");
321 }
322 }
323 }
324
325 process_array_or_scalar_values(map, path, [&](const auto& vals) -> auto {
326 auto ptr = Provisioner::new_String(vals.size(), format);
327 output.reset(ptr);
328
329 if (format == StringVector::NONE) {
330 extract_strings(vals, ptr, [](const std::string&) -> void {}, path);
331 } else if (format == StringVector::DATE) {
332 extract_strings(vals, ptr, [&](const std::string& x) -> void {
333 if (!is_date(x)) {
334 throw std::runtime_error("dates should follow YYYY-MM-DD formatting in '" + path + ".values'");
335 }
336 }, path);
337 } else if (format == StringVector::DATETIME) {
338 extract_strings(vals, ptr, [&](const std::string& x) -> void {
339 if (!is_rfc3339(x)) {
340 throw std::runtime_error("date-times should follow the Internet Date/Time format in '" + path + ".values'");
341 }
342 }, path);
343 }
344
345 extract_names(map, ptr, path);
346 return ptr;
347 });
348
349 } else if (type == "list") {
350 const auto& vals = extract_array(map, "values", path);
351 auto ptr = Provisioner::new_List(vals.size());
352 output.reset(ptr);
353 for (size_t i = 0; i < vals.size(); ++i) {
354 ptr->set(i, parse_object<Provisioner>(vals[i].get(), ext, path + ".values[" + std::to_string(i) + "]", version));
355 }
356 extract_names(map, ptr, path);
357
358 } else {
359 throw std::runtime_error("unknown object type '" + type + "' at '" + path + ".type'");
360 }
361
362 return output;
363}
